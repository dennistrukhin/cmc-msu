2.6 Выписать НАМ, не меняющий входное слово (при любом алфавите A)

|->      # Пустая строка переходит в пустую строку, и выполнение останавливается


2.8 A={ | }. Считая слово P записью положительного числа в единичной системе
счисления, уменьшить это число на 1.

|| |-> |     # Заменяем две черты на одну и останавливаемся.
|  |->       # А если изначально была только одна черта, то заменяем её на
             # пустую строку и останавливаемся
   |-> E     # Если изначально была пустая строка, выводим ошибку (Е), так как
             # не можем получит отрицательное число


2.17 A={a,b,c}. За первым символом непустого слова P вставить символ c.

Xa |-> ac    # Уберём поставленный символ, после символа а поставим с и останвимся
Xb |-> bc    # То же самое для случаев, если первым символом в изачальном слове
Xc |-> cc    # были b или c
    -> X     # Поставим в начале непустого слова символ Х


2.27 A={a,b}. Приписать справа к слову P столько палочек, сколько всего символов
входит в P (например: babb → babb||||).

|a -> a|     # Палочка ползёт вправо, пока не дойдёт до конца слова или
|b -> b|     # других уже "доползших" палочек
*a -> a*|    # Если у нас в начале слова стоит звездочка, то добавляем палочку
*b -> b*|    # и вдвигаем звездочку внутрь слова, отмечая так "посчитанные"
             # символы
* |->        # Стираем звёздочку и останавливаемся
   -> *      # Допишем в начале слова *


2.50 A={a,b}. Перенести последний символ непустого слова P в начало слова.

*a -> a*     # перегоним звёздочку до конца P
*b -> b*
a* -> A      # Когда звёхдочка в конце слова, уберем её и подменим предыдущий
b* -> B      # символ на соответствующий ему большой
aA -> Aa     # А теперь наш большой симвло движется до упора влево
aB -> Ba
bA -> Ab
bB -> Bb
A |-> a      # Большой симвло превращается в маленький, алгоритм останавливается
B |-> b
   -> *      # Поставим звёздочку в начало слова


2.64 A={( , )}. Определить, сбалансировано ли слово P по круглым скобкам.
Ответ: Д (да) или Н (нет)

() ->        # Удаляем пары скобок, в итоге. если слово сбалансировано,
             # у нас останется пустая строка
             # Если слово не сбалансированно, то останется какое-то количество
             # скобок.
  |-> Д      # Если у нас осталась пустая строка, то - ответ Да
(  -> .      # Если у нас не пустая строка, были лишние скобки
)  -> .      # Преобразуем все скобки в точки
.. -> .      # Скобок не осталось, уберём все точки, кроме одной
. |-> Н      # Получаем ответ Нет и останавливаемся
